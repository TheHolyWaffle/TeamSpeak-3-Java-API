029import java.util.Map;
030
031import com.github.theholywaffle.teamspeak3.api.ChannelProperty;
032
033public class ChannelInfo extends ChannelBase {
034
035        private final int channelId;
036
037        public ChannelInfo(int channelId, Map<String, String> map) {
038                super(map);
039                this.channelId = channelId;
040        }
041
042        @Override
043        public int getId() {
044                return channelId;
045        }
046
047        public String getDescription() {
048                return get(ChannelProperty.CHANNEL_DESCRIPTION);
049        }
050
051        public String getPassword() {
052                return get(ChannelProperty.CHANNEL_PASSWORD);
053        }
054
055        public int getCodecLatencyFactor() {
056                return getInt(ChannelProperty.CHANNEL_CODEC_LATENCY_FACTOR);
057        }
058
059        public boolean isEncrypted() {
060                return !getBoolean(ChannelProperty.CHANNEL_CODEC_IS_UNENCRYPTED);
061        }
062
063        public boolean hasUnlimitedClients() {
064                return getBoolean(ChannelProperty.CHANNEL_FLAG_MAXCLIENTS_UNLIMITED);
065        }
066
067        public boolean hasUnlimitedFamilyClients() {
068                return getBoolean(ChannelProperty.CHANNEL_FLAG_MAXFAMILYCLIENTS_UNLIMITED);
069        }
070
071        public boolean hasInheritedMaxFamilyClients() {
072                return getBoolean(ChannelProperty.CHANNEL_FLAG_MAXFAMILYCLIENTS_INHERITED);
073        }
074
075        public String getFilePath() {
076                return get(ChannelProperty.CHANNEL_FILEPATH);
077        }
078
079        public boolean isForcedSilence() {
080                return getBoolean(ChannelProperty.CHANNEL_FORCED_SILENCE);
081        }
082
083        public String getPhoneticName() {
084                return get(ChannelProperty.CHANNEL_NAME_PHONETIC);
085        }
086}
